class Drink:
    def __init__(self, name, sugar="正常甜", ice="正常冰"):
        self.name = name
        self.sugar = sugar
        self.ice = ice

    def label(self):
        return f"{self.name} / {self.sugar} / {self.ice}"

# 用 list 裝多個物件
orders = [
    Drink("珍奶", sugar="半糖", ice="少冰"),
    Drink("四季春", sugar="無糖", ice="去冰"),
    Drink("紅茶"),  # 用預設值
]

for i, drink in enumerate(orders, 1):
    print(f"第 {i} 杯：{drink.label()}")

# 修改特定訂單
orders[1].sugar = "微糖"
print(orders[1].label())  # 四季春 / 微糖 / 去冰
self 是什麼？
代表「這個物件自己」。Python 呼叫方法時自動把物件傳進去，所以第一個參數固定是 self，呼叫時不需要自己傳。

錯誤處理

常見錯誤
錯誤原因
ValueError型別轉換失敗，如 int("abc")
NameError變數未定義就使用
IndexErrorlist index 超出範圍
KeyErrordict key 不存在
TypeError型別不相容，如 "3" + 3
ZeroDivisionError除以零
try / except
try:
    a = int(input("輸入數字："))
    print(f"你輸入了：{a}")
except ValueError:
    print("這不是數字！")
except Exception as e:
    print(f"其他錯誤：{e}")
finally:
    print("不管有沒有錯都執行")
活用範例：讓程式不會因錯誤輸入而崩潰
def safe_divide(a, b):
    try:
        return a / b
    except ZeroDivisionError:
        print("錯誤：不能除以 0")
        return None

print(safe_divide(10, 2))   # 5.0
# safe_divide(10, 0) 會先印出「錯誤：不能除以 0」，再回傳 None
print(safe_divide(10, 0))   # None
